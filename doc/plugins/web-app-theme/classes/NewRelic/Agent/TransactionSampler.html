166:     def harvest(previous = nil, slow_threshold = 2.0)
167:       result = []
168:       previous ||= []
169:       
170:       previous = [previous] unless previous.is_a?(Array)
171:       
172:       previous_slowest = previous.inject(nil) {|a,ts| (a) ? ((a.duration > ts.duration) ? a : ts) : ts}

175:         
176:         if @random_sampling

178:

181:           end
182:         end
183:         
184:         slowest = @slowest_sample
185:         @slowest_sample = nil
186:         
187:         if slowest && slowest != @random_sample && slowest.duration >= slow_threshold
188:           if previous_slowest.nil? || previous_slowest.duration < slowest.duration
189:             result << slowest
190:           else
191:             result << previous_slowest
192:           end
193:         end

196:       end

This is called when we are done with the transaction. We‘ve unwound the stack to the top level.

101:

108:       @samples_lock.synchronize do
