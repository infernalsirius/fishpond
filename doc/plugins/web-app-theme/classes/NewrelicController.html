113:     @filename = params[:file]
114:     line_number = params[:line].to_i
115:     
116:     if !File.readable?(@filename)
117:       @source="<p>Unable to read #{@filename}.</p>"
118:       return
119:     end
120:     begin
121:       file = File.new(@filename, 'r')
122:     rescue => e
123:       @source="<p>Unable to access the source file #{@filename} (#{e.message}).</p>"
124:       return
125:     end
126:     @source = ""
127:     
128:     @source << "<pre>"
129:     file.each_line do |line|
130:       # place an anchor 6 lines above the selected line (if the line # < 6)
131:       if file.lineno == line_number - 6
132:         @source << "</pre><pre id = 'selected_line'>"
133:         @source << line.rstrip
134:         @source << "</pre><pre>"
135:         
136:         # highlight the selected line
137:       elsif file.lineno == line_number
138:         @source << "</pre><pre class = 'selected_source_line'>"
139:         @source << line.rstrip
140:         @source << "</pre><pre>"
141:       else
142:         @source << line
143:       end
144:     end
145:   end
